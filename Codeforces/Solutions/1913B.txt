#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t-->0){
			string s;
            cin>>s;
            int ans = 0;
            int countOne =0;
            int countZero = 0;
            for(int i =0;i<s.length();i++){
                if(s[i]=='1'){
                    countOne++;
                } else{
                    countZero++;
                }
            }
            int tLen = 0;
            for(int i =0;i<s.length();i++){
                if(s[i]=='1'){
                    if(countZero==0){
                        break;
                    } else{
                        tLen++;
                        countZero--;
                    }
                } else{
                    if(countOne==0){
                        break;
                    } else{
                       tLen++;
                        countOne--;
                    }
                }
            }
            cout<<s.length()-tLen<<endl;
	}
	return 0;

}
